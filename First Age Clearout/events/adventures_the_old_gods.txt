# An ambitious landless claimant begins to plan an adventure for the title

	id = TOG.1200
	
	hide_window = yes # Only fires for unplayable landless characters
	
	min_age = 20
	capable_only = yes
	prisoner = no
	
	trigger = {
		has_dlc = "The Old Gods"
		
		NOT = {
			has_game_rule = {
				name = adventurers
				value = none
			}
		}

		OR = {
			is_female = no
			liege = { has_law = status_of_women_4 }

				name = gender
				value = all

		}
		
		is_ruler = no

		NOT = { any_spouse = { is_ruler = yes } }

		any_claim = {
			has_holder = yes
			holder_scope = {
				independent = yes # Only do this vs independent rulers
				NAND = {
					government = nomadic_government
					OR = {
						clan_title = {
							title = PREVPREV
						}
						primary_title = {
							title = PREVPREV
						}
					}
				}
				OR = {
					# Don't target my liege unless I am ambitious
					NOT = { is_vassal_or_below = ROOT }
					ROOT = { trait = ambitious }
				}
				OR = {
					is_merchant_republic = no
					NOT = {
						capital_holding = {
							de_jure_liege_or_above = PREVPREV
						}
					}
				}
				OR = {
					# elves shouldn't be killing their own kind
					NOT = { religion_group = religion_group_eldar }
					NOT = { religion_group = ROOT }
					has_game_rule = {
						name = elf_war_rule
						value = sandbox
					}
					AND = {
						# kinslaying elves can kill other subraces of elves
						ROOT = { trait = kinslayer }
						NOT = { religion = ROOT }
					}
					# it's okay if one of them is evil
					trait = evil_side
					ROOT = { trait = evil_side }
				}
			}
			OR = {
				tier = KING
				holder_scope = { tier = PREV }
			}
		}
		
		NOT = { # Somewhat clunky way of checking that the character does not stand to inherit any titles
			any_heir_title = {
				always = yes
			}
		}
		
		diplomacy = 7
		
		health = 4
		NOT = { trait = content }
		NOT = { trait = craven }
		NOT = { trait = weak }
		
		NOT = { has_character_modifier = planning_claimant_adventure }
	}
	
	mean_time_to_happen = {
		years = 50
		modifier = {
			factor = 0.2
			has_character_flag = flag_denied_title
		}
		modifier = {
			factor = 0.5
			trait = ambitious
		}
		modifier = {
			factor = 0.2
			trait = oath_of_feanor
		}

			factor = 0.5

			factor = 0.75

			trait = brave
		}
		modifier = {
			factor = 0.75
			trait = strong
		}
		modifier = {
			factor = 0.85
			trait = proud
		}
		modifier = {
			factor = 0.85
			trait = great_elf
		}
		modifier = {
			factor = 1.5
			trait = slothful
		}
		modifier = {
			factor = 1.5
			trait = humble
		}
		modifier = {
			factor = 2.0
			trait = kind
		}
		modifier = {
			factor = 50
			has_game_rule = {
				name = adventurers
				value = rare
			}
		}
	}
	
	option = {
		random_claim = {
			limit = {
				has_holder = yes
				holder_scope = {
					independent = yes # Only do this vs independent rulers
					NAND = {
						government = nomadic_government
						OR = {
							clan_title = {
								title = PREVPREV
							}
							primary_title = {
								title = PREVPREV
							}
						}
					}
					OR = {
						# Don't target my liege unless I am ambitious
						NOT = { is_vassal_or_below = ROOT }
						ROOT = { trait = ambitious }
					}
					OR = {
						is_merchant_republic = no
						NOT = {
							capital_holding = {
								de_jure_liege_or_above = PREVPREV
							}
						}
					}
					OR = {
						# elves shouldn't be killing their own kind
						NOT = { religion_group = religion_group_eldar }
						NOT = { religion_group = ROOT }

							name = elf_war_rule
							value = sandbox
						}

							# kinslaying elves can kill other subraces of elves
							ROOT = { trait = kinslayer }
							NOT = { religion = ROOT }
						}
						# it's okay if one of them is evil
						trait = evil_side
						ROOT = { trait = evil_side }
					}
				}
				OR = {
					tier = KING
					holder_scope = { tier = PREV }
				}
			}
			
			holder_scope = {
				reverse_opinion = {
					who = ROOT
					years = 3
					modifier = opinion_claimant_adventure_target
				}
				character_event = {
					id = TOG.1201 # Warn the target
				}
			}
			ROOT = {
				add_character_modifier = {
					name = planning_claimant_adventure
					hidden = yes
					days = 730
				}
				character_event = {
					id = TOG.1202 # Fire the attack
					days = 730
				}
			}
			
			if = {
				limit = {
					holder_scope = {
						is_vassal_or_below = ROOT
					}
				}
				ROOT = { banish = yes } # Can't prepare for the attack from within the realm
			}
		}

		#TODO end_inaccessibility_effect = yes #removes in hiding or in_seclusion
		
		set_character_flag = do_not_disturb
		
		character_event = { id = TOG.1215 days = 730 }
	}
}
