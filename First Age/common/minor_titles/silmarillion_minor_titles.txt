# High Marchwarden
title_marchwarden = {
	is_high_prio = yes
	grant_limit = 4
	attribute = martial
	dignity = 0.5
	opinion_effect = 10
	realm_in_name = yes
	show_as_title = yes
	
	monthly_salary = 0.5
	monthly_prestige = 1.00
	
	allowed_to_hold = {
	
		# is either a skilled warrior or a part of the marchwardens
		OR = {
			martial = 15
			trait = great_elf
			trait = legendary_warrior
			#society_member_of = marchwardens
		}
	
		primary_title = {
			temporary = no
			OR = {
				tier = duke
				tier = king
				tier = count
				tier = baron
			}
		}
		liege = { character = FROM }
		is_adult = yes
		
		prisoner = no
		NOT = { trait = incapable }
		NOT = { trait = in_hiding }
		NOT = { has_character_flag = guru }
	}
	allowed_to_grant = {
		primary_title = {
			OR = {
				tier = king
				tier = emperor
			}
		}
		culture_group = culture_group_umanyar
	}
	
	gain_effect = {
		add_trait=marchwarden
	}
	lose_effect = {
		remove_trait=marchwarden
	}
	
	message = yes
}
