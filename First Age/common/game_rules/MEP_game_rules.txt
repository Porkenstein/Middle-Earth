	    name = canon
		text = "RULE_OPTION_CANON"
		desc = "RULE_ELF_WAR_CANON_DESC"

	    name = sandbox

	}
}

coming_of_men_rule = { # when the edain, forodwaith, and baradhrim arrive from Eriador
    name = "RULE_COMING_OF_MEN"
	option = {
	    name = canon # the forodwaith arrive around 4400, the edain around 4500, and the baradhrim around 4600
		text = "RULE_OPTION_CANON"
		desc = "RULE_COMING_OF_MEN_CANON_DESC"
	}
	option = {
	    name = sandbox # the different groups of men arrive at random times between 4350 and 4800
		text = "RULE_OPTION_SANDBOX"
		desc = "RULE_COMING_OF_MEN_SANDBOX_DESC"
	}
	option = {
	    name = off # men never come from the east
		text = "RULE_OPTION_OFF"
		desc = "RULE_COMING_OF_MEN_OFF_DESC"
	}
}
