obj_become_a_craftsman = {
	type = character
	allow = {
		is_female = no
		OR = {
			culture = culture_noldor
			culture_group = culture_group_dwarves
		}
		is_adult = yes
		stewardship = 7
		age = 60
		NOT = { trait = incapable }
		NOT = { trait = weak }
		NOT = { trait = imbecile }
		NOT = { trait = slow }
		## Not a craftsman ##
		NOT = { trait = craftsman }
		NOT = { trait = skilled_craftsman }
		NOT = { trait = master_craftsman }
		NOT = { trait = legendary_craftman }
		## Not an herbalist ##
		NOT = { trait = herbalist }
		NOT = { trait = apothecary }
		NOT = { trait = great_healer }
		NOT = { trait = healer_master }	
		## Not a Musician ##
		NOT = { trait = musician }
		NOT = { trait = composer }
		NOT = { trait = skilled_composer }
		NOT = { trait = legendary_musician }	
		## Not a Mariner ##
		NOT = { trait = mariner }
		NOT = { trait = navigator }
		NOT = { trait = captain }
		NOT = { trait = ship_builder } 
		NOT = { has_character_flag = obj_become_a_craftsman }
		NOT = { has_character_flag = not_a_craftsman }
	}
	chance = {
		factor = 100

		modifier = {
			factor = 2
			trait = honorable
		}
		modifier = {
			factor = 0.5
			is_female = yes
		}
		modifier = {
			factor = 2.0
			trait = ambitious
		}
		modifier = {
			factor = 2.0
			trait = proud
		}
		modifier = {
			factor = 2.0
			OR = {
				trait = strong
				trait = tall
			}
		}
		modifier = {
			factor = 2.0
			trait = duelist
		}
		modifier = {
			factor = 1.5
			OR = {
				trait = tough_soldier
				trait = skilled_tactician
				trait = brilliant_strategist
			}
		}
	}
	success = {
		OR = { 
			trait = craftsman
			trait = skilled_craftsman
			trait = master_craftsman
			trait = legendary_craftman			
		}
	}
	abort = {
		OR = {
			has_character_flag = abort_obj
			has_character_flag = not_a_craftsman
			trait = incapable
			NOT = { stewardship = 5 }
		}
	}
	abort_effect = {
		clr_character_flag = abort_obj
		clr_character_flag = obj_check
	}
	effect = {
		prestige = 10
		set_character_flag = obj_become_a_craftsman
	}
}
