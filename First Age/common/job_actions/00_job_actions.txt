# Chancellor
action_improve_relations = {
	attribute = diplomacy
	
	potential = {
	}
	
	trigger = {
		NOT = { religion = religion_wilderness }
		any_province_lord = {
			NOT = { character = FROM }
			FROM = {
				job_chancellor = {
					NOT = { character = PREVPREV } # Can't be sent to improve relations with himself...
				}
			}
		}
		is_within_diplo_range = FROM
	}

	events = { 910 911 }
}

	}

}

# Court Chaplain
action_inquisition = { # enlighten dwarves and men
	attribute = learning
	
	potential = {
		FROM = {
			NOT = { trait = evil_side }
			religion_group = religion_group_eldar
			NOT = { is_tribal = yes }
			
			job_spiritual = {
				NOT = {
					has_character_modifier = in_seclusion
				}
			}
		}
	}
	
	trigger = {
		NOT = { religion = religion_wilderness }
		num_of_settlements = 1

		owner = {
			is_within_diplo_range = FROM
			NOT = { religion_group = religion_group_melkor }
			NOT = { religion_group = religion_group_eldar }
			top_liege = {
				NOT = { religion_group = religion_group_eldar }
				NOT = { religion_group = religion_group_melkor }
			}

			OR = {
				# evil ruler
				trait = evil_side

				# not already an elf-friend
				AND = {
					NOT = { religion_group = religion_group_eldar }
					NOT = { trait = elf_friend }
				}
			}
		}
	}

	events = { 903 }
}

# search for resources in a province
action_survey = {
	attribute = learning

	potential = {
		FROM = {
			culture_group = culture_group_dwarves
			job_spiritual = {
				NOT = {
					has_character_modifier = in_seclusion
				}
			}
		}
	}
	trigger = {
		is_land = yes
		OR = {
			# dwarf-lords with less clout can't survey anywhere in the world
			owner = { is_within_diplo_range = FROM }
			FROM = { tier = king }
			FROM = { tier = emperor }
			
		}
		OR = {
			FROM = { ai = no }
			has_hidden_resources = yes
		}
	}

	events = { }
}
