### Berserker Events

# Character becomes a Berserker in battle
character_event = {
	id = TOG.3000
	desc = EVTDESC_TOG_3000
	picture = GFX_evt_viking_battle_oldgods
	border = GFX_event_normal_frame_war

	is_triggered_only = yes

	trigger = {
		has_dlc = "The Old Gods"

		is_alive = yes
		is_female = no
		NOT = { trait = blood_thirsty }
		OR = {
			trait = wroth
			trait = evil_side
			trait = lunatic
		}
	}

	option = {
		name = EVTOPTA_TOG_3000
		add_trait = blood_thirsty
	}
}

# Berserker is Killed in battle
character_event = {
	id = TOG.3003
	desc = EVTDESC_TOG_3003
	picture = GFX_evt_viking_battle_oldgods
	border = GFX_event_normal_frame_war

	is_triggered_only = yes

	trigger = {
		has_dlc = "The Old Gods"
		trait = blood_thirsty
	}

	immediate = {
		death = {
			death_reason = death_battle
			killer = FROMFROM
		}
		FROMFROM = {
			character_event = { id = 249 }
			set_character_flag = temp_battle_kill_242
		}
		FROM = {
			if = {
				limit = { NOT = { has_character_flag = temp_battle_kill_242 } }
				character_event = { id = 249 }
			}
		}
		FROMFROM = {
			clr_character_flag = temp_battle_kill_242
		}
	}

	option = {
		name = EVTOPTA_TOG_3003
	}
}

# Berserker goes berserk against Courtier (Hidden AI)
character_event = {
	id = TOG.3009
	hide_window = yes

	ai = yes
	min_age = 16
	only_men = yes
	capable_only = yes
	prisoner = no

	trigger = {
		has_dlc = "The Old Gods"
		host = {
			character = liege
			prisoner = no
			trait = blood_thirsty
			NOT = {
				trait = incapable
				opinion = { who = ROOT value = 10 }
			}
			NOT = { trait = in_hiding }
		}
		NOT = { trait = blood_thirsty }
		is_primary_heir = no
		is_pretender = no
	}

	mean_time_to_happen = {
		months = 2080
	}

	immediate = {
		liege = {
			character_event = { id = TOG.3006 }
		}
	}
}

# Great Ork Trait Gained
character_event = {
	id = TOG.3320
	desc = EVTDESC_TOG_3320
	picture = GFX_evt_vikings_arriving_oldgods
	border = GFX_event_normal_frame_war

	hide_from = yes

	is_triggered_only = yes

	min_age = 16
	capable_only = yes
	prisoner = no

	trigger = {
		has_dlc = "The Old Gods"
		NOT = {
			trait = great_orc
			trait = impaler
		}
		religion = religion_orkish
	}

	option = {
		name = EVTOPTA_TOG_3320
		remove_trait = weak_orc
		remove_trait = orc_warrior
		remove_trait = strong_orc
		add_trait = great_orc
		add_trait = impaler
	}
