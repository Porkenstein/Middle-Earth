namespace = racetrait

## The characters is an Elf ##
character_event = {
	id = racetrait.1
	
	hide_window = yes
	
	trigger = {
		religion_group = religion_group_eldar
		NOT = { trait = elf }
		NOT = { trait = half_elf }
	}

	mean_time_to_happen = {
		days = 7
	}

	option = {
		add_trait = elf
		if = {
			limit = {
				ROOT = { NOT = { age = 45 }	}
			}
			ROOT = {
				character_event = { id = elfchild.0 }
			}
		}
	}
}

## The characters is an Orc, but don't have any of the orcs warriors traits at beginning ##
character_event = {
	id = racetrait.10
	
	hide_window = yes
	
	trigger = {
		religion = religion_orkish
		NOT = { trait = weak_orc }
		NOT = { trait = orc_warrior }
		NOT = { trait = strong_orc }
		NOT = { trait = great_orc }
		age = 16
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
	    random_list = {
			70 = {
				add_trait = weak_orc
			}
			15 = {
				add_trait = orc_warrior
			}
			10 = {
				add_trait = strong_orc
			}
			5 = {
				add_trait = great_orc
			}
		}
	}
}
