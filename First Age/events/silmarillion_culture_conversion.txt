#--------------------------------------
# The Golodhrim Melting Pot
#--------------------------------------
character_event = {
	id = culture_conversion.0
	desc = "EVTDESC_culture_conversion_0"
	picture = GFX_evt_stone_church
	
	only_playable = yes
	culture = culture_noldor
	
	trigger = {
		should_evolve_cultures_trigger = yes
		year = 4400
		higher_tier_than = DUKE
		NOT = {
			# not nos feanor
			dynasty = 3007
		}
		any_realm_province = {
			region = world_hithlum
			culture = culture_mithrim
		}
		any_realm_province = {
			region = world_hithlum
			culture = culture_noldor
		}
	}

	mean_time_to_happen = {
		months = 1200

		modifier = {
			factor = 0.2
			OR = {
				has_game_rule = {
					name = culture_conversion
					value = faster_melting_pots
				}
				has_game_rule = {
					name = culture_conversion
					value = faster_melting_pots_and_slower_normal
				}
			}
		}
		modifier = {
			factor = 0.5
			independent = yes
		}
		modifier = {
			factor = 2.0
			NOT = { learning = 11 }
		}
	}

	option = {
		name = "EVTOPTA_culture_conversion_0"
		culture = culture_golodhrim
		random_demesne_province = {
			limit = {
				NOT = { culture = culture_golodhrim }
				region = world_hithlum
				# only start replacing the noldor culture (at first)
				culture = culture_noldor
			}
			culture = culture_golodhrim
		}
	}
}

		year = 4400

				}
