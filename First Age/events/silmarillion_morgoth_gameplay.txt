namespace = morgothevents

# Evil creatures should never go bankrupt, although mobilizing a great orc host should justifiably drain wealth.
character_event = {
	id = morgothevents.1
	hide_window = yes
	trigger = {
		NOT = { wealth = 0 }
		OR = {
			religion_group = religion_group_melkor
			religion = religion_eruhini
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		wealth = 10000
		clear_wealth = yes
	}
}
