## Rebuilding of Iant Iaur Events ##
character_event = {
	id = buildings.20
	title = "EVTTITbuildings.20"
	desc = "EVTDESCbuildings.20"
	picture = GFX_evt_rebuilding

	is_triggered_only = yes
	
	option = {
		name = "EVTOPTAbuildings.20" ## I'll rebuild Esgaliant
		set_global_flag = iant_iaur_rebuilt
		wealth = -50
		FROM = {
			character_event = { id = buildings.21 days = 30 } # Rebuilding of Esgaliant - First Step
		}
	}	
	option = {
		name = "EVTOPTBbuildings.20"
		trigger = { ai = no }
	}		
}

## Rebuilding of Iant Iaur - First Step ##
character_event = {
	id = buildings.21
	title = "EVTTITbuildings.21"
	desc = "EVTDESCbuildings.21"
	picture = GFX_evt_rebuilding

	is_triggered_only = yes
	
	option = {
		name = "EVTOPTAbuildings.21" ## I need to kill the spiders.
		wealth = -20
		random_list = {
			70 = {
				FROM = {
					custom_tooltip = {
						text = dark_terrors_slain_tt
						character_event = { id = buildings.22 days = 30 } # Rebuilding of Iant Iaur - Second Step
					}
				}
				56 = { # Iant Iaur
					remove_province_modifier = infested # dark terrors removed
				}
				modifier = {
					factor = 0.5
					NOT = { martial = 15 }
				}
				modifier = {
					factor = 0
					NOT = { martial = 5 }
				}
			}
			30 = {
				FROM = {
					custom_tooltip = {
						text = dark_terrors_slain_failure_tt
						character_event = { id = buildings.23 days = 45 } # First Fail
					}
				}
			}			
		}
	}	
	option = {
		name = "EVTOPTBbuildings.21"
		trigger = { ai = no }
		clr_global_flag = iant_iaur_rebuilt
	}		
}

## Rebuilding of Iant Iaur Events - Success ##
character_event = {
	id = buildings.24
	title = "EVTTITbuildings.24"
	desc = "EVTDESCbuildings.24"
	picture = GFX_evt_rebuilding

	is_triggered_only = yes
	
	option = {
		name = "EVTOPTAbuildings.24" ## Great !
		prestige = 150
		set_global_flag = iant_iaur_rebuilt_finally		
		56 = { # Iant Iaur
			build_holding = { title = b_iant_iaur type = castle holder = ROOT }
			add_wonder_stage = 1
		}
		add_trait = dwarf_friend
		add_trait = architect
		if = {
			limit = {
				has_dlc = "Reapers"
			}
			custom_tooltip = {
				text = increase_prosperity_custom_tooltip
				hidden_tooltip = {
					56 = { # Iant Iaur
						change_variable = { which = prosperity_value value = 6 }
					}
				}
			}
		}
	}		
}
