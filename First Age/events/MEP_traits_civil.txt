#### Learning to heal from your guardian #####
character_event = {
	id = civiltraits.1
	desc = "EVTDESCciviltraits.1"
	picture = GFX_evt_child_sword
	
	min_age = 6
	max_age = 16

	trigger = {
		guardian = {
			OR = {
				trait = herbalist
				trait = apothecary
				trait = great_healer
				trait = healer_master
			}
		}
		NOT = { trait = herbalist }
		NOT = { trait = apothecary }
		NOT = { trait = great_healer }
		NOT = { trait = healer_master }
	}

	mean_time_to_happen = {
		months = 30

		# Mentor Effects
		modifier = {
			factor = 0.85
			OR = {
				guardian = { trait = herbalist }
				guardian = { trait = apothecary }
				guardian = { trait = great_healer }
				guardian = { trait = healer_master }
			}
		}

		# Personality Effects
		#Good
		modifier = {
			factor = 0.85
			OR = {
				trait = patient
				trait = charitable
			}
		}
		#Bad
		modifier = {
			factor = 1.35
			OR = {
				trait = slothful
				trait = craven
			}
		}		
		
		# Intelligence Effects
		#Good
		modifier = {
			factor = 0.85
			trait = genius
		}
		modifier = {
			factor = 0.9
			trait = quick
		}
		#Bad
		modifier = {
			factor = 1.2
			trait = slow
		}
		modifier = {
			factor = 1.35
			trait = imbecile
		}
		
		# Physical Effects
		#Good
		modifier = {
			factor = 0.85
			trait = kind
		}
		#Bad
		modifier = {
			factor = 2
			trait = cruel
		}				
	}

	option = {
		name = "EVTOPTAciviltraits.1" # Stick 'em with the pointy end!
		hidden_tooltip = { guardian = { character_event = {id = civiltraits.5} } } #Guardian Notification.
		add_trait = herbalist 
	}
}
