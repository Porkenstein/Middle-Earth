	id = 39600
	desc = EVTDESC39600
	picture = GFX_evt_heretic
	border = GFX_event_normal_frame_religion
	
	is_triggered_only = yes
	trigger = {
		NOT = { has_character_flag = flag_converting_baptism }
		OR = {
			ai = yes
			FROMs_religion_is_playable_trigger = yes
		}
	}
	
	fail_trigger_effect = {
		FROM = {
			clr_character_flag = preacher_arrives
		}
	}
	
	option = {
		name = EVTOPTA39600 # Throw the preacher in jail
		ai_chance = {
			factor = 3
			# Always allow them if you already like that religion
			modifier = {
				factor = 0
				trait = elf_friend
			}
		}
		
		FROM = {
			imprison = ROOT
			liege = {
				set_character_flag = ai_failed_mission # AI flag - used in the code
				opinion = {
					modifier = opinion_imprisoned_missionary
					who = ROOT
				}
				hidden_tooltip = {
					reverse_opinion = {
						modifier = opinion_sent_preacher
						who = ROOT
					}
					character_event = { id = 39605 days = 1 }
				}
			}
		}
		custom_tooltip = { text = preacher_imprisoned }
	}
	option = {
		name = EVTOPTB39600 # Allow them to preach
		ai_chance = {
			factor = 1
			
			# Traits
			modifier = {
				factor = 0.3
				trait = zealous
			}
			modifier = {
				factor = 0.3
				NOT = { opinion = { who = FROM value = -10 } }
			}
			modifier = {
				factor = 3
				trait = cynical
			}
			modifier = {
				factor = 2
				trait = arbitrary
			}
			# Personal
			modifier = {
				factor = 0.3
				FROM = {
					is_rival = ROOT
				}
			}
			modifier = {
				factor = 3
				OR = {
					any_spouse = {
						trait = elf_friend
						reverse_opinion = { who = ROOT value = 50 }
					}
					any_consort = {
						trait = elf_friend
						reverse_opinion = { who = ROOT value = 50 }
					}
					any_lover = {
						trait = elf_friend
						reverse_opinion = { who = ROOT value = 50 }
					}
					any_child = {
						trait = elf_friend
						reverse_opinion = { who = ROOT value = 50 }
					}
				}
			}
			
			# Political
			modifier = {
				factor = 0.1
				any_war = {
					using_cb = evil_invasion
					attacker = {
						character = ROOT
					}
				}
			}
			modifier = {
				factor = 4
				any_war = {
					using_cb = good_invasion
					defender = {
						character = ROOT
					}
				}
			}
			modifier = {
				factor = 2
				any_realm_province = {
					religion = FROM # listening to your folk
				}
			}
			
			# Pragmatism
			modifier = {
				factor = 2
				any_realm_province = {
					any_neighbor_province = {
						owner = {
							top_liege = {
								trait = good_side
								relative_power = { who = ROOT power = 2.0 } # Powerful good neighbor
							}
						}
					}
				}
			}
		}
		
		FROM = {
			liege = {
				opinion = {
					modifier = opinion_allowed_missionary
					who = ROOT
				}
			}
			hidden_tooltip = {
				character_event = { id = 39612 days = 1 }
			}
		}
		custom_tooltip = { text = preacher_allowed }
	}
}
