		culture_group = culture_group_dwarves

		}

	option = {
		name = "EVTOPTBmepmarriage.0"#Actually, I kind of like her...

	}	
}

# Simple infertility events
# hidden event that dispatches mepmarriage.14 or just adds not_fertile if conditions are correct
character_event = {
	id = mepmarriage.13
	hide_window = yes
	min_age = 45
	is_triggered_only = yes

	trigger = {
		NOT = { mep_is_immortal_race_trigger = yes }
		NOT = { religion_group = religion_group_wilderness }
		NOT = { religion_group = religion_group_khazad }
		NOT = {
			culture_group = culture_group_dwarves
		}
		NOT = {
			has_character_modifier = mep_sinfertile
		}
		is_alive = yes
	}

	option = {
		random_list = {
			1 = {
				modifier = {
					factor = 0
					NOT = {
						has_children = yes
					}
				}
				modifier = {
					factor = 0.1
					NOT = {
						any_child = {
							is_female = yes
						}
					}
				}
				modifier = {
					factor = 2
					has_children = yes
				}
				modifier = {
					factor = 2
					num_of_children = 2
				}
				modifier = {
					factor = 10
					num_of_children = 3
				}
				character_event = {
					id = mepmarriage.131
				}
			}
			1 = {
			}
		}
	}
}
character_event = {
	id = mepmarriage.131
	hide_window = yes

	is_triggered_only = yes

	trigger = {
		NOT = { mep_is_immortal_race_trigger = yes }
		NOT = { religion_group = religion_group_wilderness }
		NOT = { religion_group = religion_group_khazad }
		NOT = {
			has_character_modifier = mep_sinfertile
		}
		is_alive = yes
	}

	option = {# option for Elven/Númenorean blood
		trigger = {
			mep_is_human_trigger = yes
		}
		IF = {
			limit = {
				OR = {
					AND = {
						is_female = yes
						age = 65
					}
					AND = {
						is_female = no
						age = 85
					}
				}
			}
			character_event = { id = mepmarriage.14 }
		}
	}

	# Half-Orcs with Elven/Númenorean Blood ?
	option = {# Option for Hobbits
		trigger = {
			religion_group = religion_group_periannath
		}
		IF = {
			limit = { age = 50 }
			add_character_modifier = {
				modifier = mep_sinfertile
				duration = -1
				hidden = yes
			}
		}
	}

	option = {# Orcs/Goblins
		trigger = {
			mep_orc_culture = yes
		}
		if = {
			limit = { age = 45 }
			add_character_modifier = {
				modifier = mep_sinfertile
				duration = -1
				hidden = yes
			}
		}
	}

	option = {# Trolls
