##These events are for when an heir to E_Gondor or E_Arnor does not have the blood of numenor traits##

namespace = newkinstrife

##King is informed that his heir is not considered suitable by many people in Gondor##
character_event = {
	id = newkinstrife.1
	desc = "EVTDESCnewkinstrife.1"#Word reaches your ears that some of your vassals of old prestigious families do not consider your heir worthy, as the blood of Numenor is clearly weak in that individual. Hopefully it won't lead to a second kinstrife upon your death...
	picture = "GFX_death"
	
		
	trigger = {
		has_landed_title = e_gondor
		trait = numenor_blood
		current_heir = {
			age = 5
			NOT = { trait = numenor_blood }
			NOT = { trait = elf }
			NOT = { trait = half_elf }
		}
		NOT = { has_character_flag = newkinstrife1 }
	}
		
	mean_time_to_happen = {
		days = 7
	}
				  
	option = {
		name = "EVTOPTAnewkinstrife.1"#How dare they!
		set_character_flag = newkinstrife1
	}
	 
	option = {
		name = "EVTOPTBnewkinstrife.1"#We don't want a second kinstrife...
		clr_character_flag = newkinstrife1
	}
}
	
##Someone takes the throne who is not numenorean blooded, ##
character_event = {

	}
				  
	option = {
		name = "EVTOPTAnewkinstrife.2"#How dare they!
		set_character_flag = newkinstrife2
		set_global_flag = newkinstrife
	}
	 
	option = {
		name = "EVTOPTBnewkinstrife.2"#They better not try anything...
		set_character_flag = newkinstrife2
		set_global_flag = newkinstrife
	}
	
}
	
##A King level vassal begins to plot and scheme##
character_event = {
	id = newkinstrife.3
	title = "EVTTITnewkinstrife.3"#You are troubled
	desc = "EVTDESCnewkinstrife.3"#The former High King is dead, but he married beneath his station, and his heir is not worthy of the throne of Elendil. Maybe there is another of the Royal House more... suitable?
	picture = "GFX_death"


	trigger = {
		OR = {
			culture = culture_gondorian
			culture = culture_arnorian
		}
		trait = numenor_blood
		NOT = { has_character_flag = newkinstrife3 }
		NOT = { has_landed_title = e_gondor }
		NOT = { has_landed_title = e_arnor  }
		has_global_flag = newkinstrife
		higher_tier_than = DUKE
	}

	mean_time_to_happen = {
		days = 1000
	}

	option = {
		name = "EVTOPTAnewkinstrife.3"#Let's see if there is...
		set_character_flag = newkinstrife3
		clr_global_flag = newkinstrife

		any_playable_ruler  = {
			limit = {
				trait = numenor_blood
				trait = blood_of_isildur
				NOT = { has_landed_title = e_gondor }
				NOT = { has_landed_title = e_arnor }
				#age = 12
			}
			character_event = { id = newkinstrife.4 days = 14 }
		}
		character_event = { id = newkinstrife.5 days = 180 }
	}

	option = {
		name = "EVTOPTBnewkinstrife.3"#No, I will not start another kinstrife!
		trigger = { ai = no }
		set_character_flag = newkinstrife3
		clr_global_flag = newkinstrife
	}
}
	
##If there is an heir, a global flag called newkinstrife2 will appear, triggering an event. If there is not, an event will trigger after##
##180 days, where the noble will put himself up for the throne##

character_event = {
	id = newkinstrife.4
	title = "EVTTITnewkinstrife.4"#You receive word
	desc = "EVTDESCnewkinstrife.4"#The former High King of Gondor is dead, but many nobles do not feel his heir worthy. You too are an heir of Kings, one more... suitable. They want you to take the throne. Do you accept?
	picture = "GFX_death"
	
	is_triggered_only = yes
	
	option = {
		name = "EVTOPTAnewkinstrife.4"#Yes, I was born to rule!
		FROM = { character_event = { id = newkinstrife.6 days = 7 } }
		set_global_flag = newkinstrife2
		set_character_flag = kinstrifeclaimant
	}
	
	option = {
		name = "EVTOPTBnewkinstrife.4"#No, I will not plot against my liege!
		trigger = { ai = no }
	}
	
}
	
##this event checks to see if there was an heir that accepted the offer - if no global flag is newkinstrife2, then there was not##

character_event = {
	id = newkinstrife.5
	desc = "EVTDESCnewkinstrife.5"#Did you find an heir of the royal house willing to take the throne?
	picture = "GFX_kinstrife"

	is_triggered_only = yes

	option = {
		name = "EVTOPTAnewkinstrife.5"#Yes, all is well!
		trigger = { has_global_flag = newkinstrife2 }
	}

	option = {
		name = "EVTOPTBnewkinstrife.5"#No, I'll have to take it myself then...
		trigger = { NOT = { has_global_flag = newkinstrife2 } }
		any_playable_ruler = {
			limit = {
				trait = numenor_blood
				higher_tier_than = DUKE
				top_liege = {
					has_landed_title = e_gondor
				}
				NOT = { has_landed_title = e_gondor }
				NOT = { has_landed_title = e_arnor }
			}
			character_event = { id = newkinstrife.15 days = 30 }
		}
	}
	
}

##the claimant makes his final move##
character_event = {
	id = newkinstrife.19
	title = "EVTTITnewkinstrife.19"#It is time.
	desc = "EVTDESCnewkinstrife.19"#You've come this far, now there is no going back. You must declare yourself openly, and let your supporters know that they are with you to the end now.
	picture = "GFX_kinstrife"

	trigger = {
		trait = numenor_blood
		has_character_flag = kinstrifeclaimthethrone
	}

	immediate = {
		wealth = 1000
		ROOT = {
			create_title = {
				tier = EMPEROR
				landless = yes
				temporary = yes
				culture = ROOT
				name = "newkinstrife_claimant"
				holder = ROOT
			}
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = "EVTOPTAnewkinstrife.19"#Do it.
		clr_character_flag = kinstrifeclaimthethrone
		e_gondor = {
			reverse_war = {
				target = ROOT
				casus_belli = claim_all
			}
		}
		any_playable_ruler = {
			character_event = { id = newkinstrife.14 days = 7 }
		}
		any_playable_ruler = {
			limit = {
				trait = numenor_blood
				has_character_flag = kinstrifeplotter
				top_liege = {
					has_landed_title = e_gondor
				}
			}
			character_event = { id = newkinstrife.12 }
		}
	}
}
