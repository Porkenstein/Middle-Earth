# Event to force Theocratic gov on temple holders
namespace = theo_event

character_event = {
    id = theo_event.1
    is_triggered_only = yes
    hide_window = yes

    trigger = {
        FROM = {
            holding_type = temple
        }
        NOT = {
            demesne_size = 2
        }
        NOT = {
            is_theocracy = yes
        }
    }

    immediate = {
        set_government_type = theocratic_government
    }
}
