character_event = {	#Maintenance
	id = mep_on_actions_mt.0
	is_triggered_only = yes
	hide_window = yes

	trigger = {
		trait = trait_old_age
		OR = {
			AND = {
				age = 16
				NOT = {
					age = 55
				}
			}
			AND = {
				age = 55
				NOT = {
					age = 120
				}
			}
			AND = {
				age = 120
				NOT = {
					age = 130
				}
			}
		}
	}

	option  = {
		remove_trait = trait_old_age
		add_trait = trait_old_age
	}
}
