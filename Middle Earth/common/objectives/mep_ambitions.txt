#### Claim the Palantiri #####
### The Anor-Stone ###	
obj_claim_palantir_anor = {
	type = character
	allow = {
		is_female = no
		culture_group = culture_group_numenorean		
		is_adult = yes
		NOT = { has_artifact = artifact_palantir_anor }
		has_landed_title = c_minas_tirith
		NOT = { has_character_flag = obj_claim_palantir_anor }
	}
	chance = {
		factor = 100
		modifier = {
			factor = 2
			trait = honorable
		}
		modifier = {
			factor = 0.5
			is_female = yes
		}
		modifier = {
			factor = 2.0
			trait = ambitious
		}
		modifier = {
			factor = 2.0
			trait = proud
		}
		modifier = {
			factor = 2.0
			OR = {
				trait = strong
				trait = tall
			}
		}
		modifier = {
			factor = 2.0
			trait = duelist
		}
	}
	success = {
		has_artifact = artifact_palantir_anor
	}
	abort = {
		OR = {
			has_character_flag = abort_obj
			trait = incapable
			NOT = { has_landed_title = c_minas_tirith }
		}
	}
	abort_effect = {
		clr_character_flag = abort_obj
		clr_character_flag = obj_check
	}
	effect = {
		prestige = 10
		set_character_flag = obj_claim_palantir_anor
	}	
}
