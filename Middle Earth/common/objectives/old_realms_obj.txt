#Silvan Realm
obj_claim_silvan = {
	type = character
	can_cancel = no

	potential = {
		is_adult = yes
		is_playable = yes
		is_landed = yes
		independent = yes
		higher_real_tier_than = BARON
		prestige = 3000
		OR = {
			culture_group = culture_group_amanyar
			culture_group = culture_group_moriquendi
			culture_group = culture_group_umanyar
		}
		NOT = {
			e_silvan_elves = {
				has_holder = yes
			}
		}
		OR = {
			has_landed_title = k_elves
			has_landed_title = k_mirkwood
			primary_title = {
				lower_real_tier_than = KING
				kingdom = {
					OR = {
						title = k_elves
						title = k_mirkwood
					}
				}
			}
		}
	}
	allow = {
		NOT = { trait = incapable }
	}
	chance = {
		factor = 30
		modifier = {
			factor = 3
			OR = {
				k_elves = {
					has_holder = no
				}
				k_mirkwood = {
					has_holder = no
				}
			}
		}
		modifier = {
			factor = 3
			trait = ambitious
		}
		modifier = {
			factor = 2
			trait = brave
		}
		modifier = {
			factor = 3
			independent = yes
		}
		modifier = {
			factor = 0.1
			trait = craven
		}
		modifier = {
			factor = 0.01
			trait = content
		}
		modifier = {
			factor = 0.01
			trait = slothful
		}
		modifier = {
			factor = 0.1
			tier = COUNT
		}
	}
	success = {
		completely_controls = k_elves
		completely_controls = k_mirkwood
	}
	abort = {
		OR = {
			e_silvan_elves = {
				has_holder = yes
				NOT = {
					holder_scope = {
						character = ROOT
					}
				}
			}
			trait = incapable
			NOT = {
				is_playable = yes
			}
			NOT = {
				is_landed = yes
			}
			NOT = {
				independent = yes
			}
			NOR = {
				culture_group = culture_group_amanyar
				culture_group = culture_group_moriquendi
				culture_group = culture_group_umanyar
			}
		}
	}
	abort_effect = {
		prestige = -200
	}
	effect = {
		prestige = 1000
		e_silvan_elves = {
			grant_title = ROOT
		}
		any_demesne_title = {
			limit = {
				real_tier = EMPEROR
				NOT = {
					title = e_silvan_elves
				}
			}
			destroy_landed_title = THIS
		}
		#Add Special Bloodline
	}
}
